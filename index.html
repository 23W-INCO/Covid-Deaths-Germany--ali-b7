Welcome to Covid Data Analysis Application

This app provides interactive visualizations of the results of the COVID-19 pandemic in Germany.

For more details, click on a federal state marker.

// Load the data from the CSV file d3.csv("german_states.csv").then(function(data) { // Parse data data.forEach(function(d) { d['Covid Cases'] = +d['Covid Cases']; // Convert to numerical value d['Total Deaths'] = +d['Total Deaths']; // Convert to numerical value d['Population'] = +d['Population']; // Convert to numerical value, assuming Population field is clean }); // Sort the data by 'Covid Cases' in ascending order data.sort(function(a, b) { return a['Covid Cases'] - b['Covid Cases']; }); // Set dimensions and margins of the graph var margin = {top: 20, right: 20, bottom: 100, left: 80}, // Increased bottom margin width = 1000 - margin.left - margin.right, height = 600 - margin.top - margin.bottom; // Adjusted for the new SVG height // Set the ranges var x = d3.scaleBand() .range([0, width]) .padding(0.1); var y = d3.scaleLinear() .range([height, 0]); // Append the svg object to the body of the page var svg = d3.select("#barGraph") .attr("width", width + margin.left + margin.right) .attr("height", height + margin.top + margin.bottom) .append("g") .attr("transform", "translate(" + margin.left + "," + margin.top + ")"); // Scale the range of the data in the domains x.domain(data.map(function(d) { return d.Bundesland; })); y.domain([0, d3.max(data, function(d) { return d['Covid Cases']; })]); // Append the rectangles for the bar chart svg.selectAll(".bar") .data(data) .enter().append("rect") .attr("class", "bar") .attr("x", function(d) { return x(d.Bundesland); }) .attr("width", x.bandwidth()) .attr("y", function(d) { return y(d['Covid Cases']); }) .attr("height", 0) // Start with height 0 .transition() // Start a transition .duration(800) // Duration of the transition in milliseconds .attr("height", function(d) { return height - y(d['Covid Cases']); }); // Transition to the actual height // Add the x Axis var xAxis = svg.append("g") .attr("class", "x axis") .attr("transform", "translate(0," + height + ")") .call(d3.axisBottom(x)); xAxis.selectAll("text") .style("text-anchor", "end") .attr("dx", "-.8em") .attr("dy", ".15em") .attr("transform", "rotate(-45)"); // Adjusted rotation angle // Add the y Axis var yAxis = svg.append("g") .attr("class", "y axis") .call(d3.axisLeft(y).tickFormat(d3.format("~s"))); yAxis.selectAll("text") .style("text-anchor", "end"); // Chart title svg.append("text") .attr("transform", "translate(100,0)") .attr("x", 50) .attr("y", 50) .attr("font-size", "24px") .text("Covid-19 cases in Germany 2020-2022") // Add the title on the left side svg.append("text") .attr("class", "title") .attr("transform", "rotate(-90)") .attr("x", -height / 2) .attr("y", -margin.left) .attr("dy", "1em") .style("text-anchor", "middle") .text("Number of Cases"); // Add a hover event listener to highlight bars on hover svg.selectAll(".bar") .on("mouseover", function(d) { d3.select(this) .classed("highlighted", true); // Apply the 'highlighted' class }) .on("mouseout", function(d) { d3.select(this) .classed("highlighted", false); // Remove the 'highlighted' class }); // Add a click event listener to toggle highlighting on click svg.selectAll(".bar") .on("click", function(d) { d3.select(this) .classed("highlighted", !d3.select(this).classed("highlighted")); // Toggle the 'highlighted' class }); function createBubbleChart() { // Load the data from the CSV file d3.csv("covid_de1.csv").then(function(data) { // Parse data data.forEach(function(d) { d.Population = +d.Population; d['Total Deaths'] = +d['Total Deaths']; }); var svgWidth = 1000, svgHeight = 600; var margin = { top: 50, right: 50, bottom: 50, left: 50 }; var width = svgWidth - margin.left - margin.right; var height = svgHeight - margin.top - margin.bottom; d3.select('#bubbleChart').selectAll("*").remove(); var svg = d3.select('#bubbleChart') .attr('width', svgWidth) .attr('height', svgHeight) .append('g') .attr('transform', 'translate(' + margin.left + ',' + margin.top + ')'); var radiusScale = d3.scaleSqrt() .domain([0, d3.max(data, function(d) { return d.Population; })]) .range([0, 50]); var colorScale = d3.scaleSequential() .domain([0, d3.max(data, function(d) { return d['Total Deaths']; })]) .interpolator(d3.interpolateReds); svg.selectAll('.bubble') .data(data) .enter().append('circle') .attr('class', 'bubble') .attr('cx', function(d, i) { return (width / data.length) * i; }) .attr('cy', height / 2) .attr('r', function(d) { return radiusScale(d.Population); }) .style('fill', function(d) { return colorScale(d['Total Deaths']); }); }); }
